# processor.py | VLDS Processing Pipeline — Forest Dealerships # ───────────────────────────────────────────────────────────────────────────── # Steps 1-10: # 1 Load VLDS Q4 2025 file # 2 LOB / Sub-LOB / Segment / AUTO eligibility # 3 DDS-Acoustics + Battery exclusion # 4 Credit Memo-Misc eligibility # 5 Exclusion list (49 criteria) # 6 DF/RECD exclusion (concat keys + Remark) # 7 Finalise eligible records # 8 Quarterly Pivot (Customer × Month, Grand Total col + row) # ── Step 2: LOB / Sub-LOB / Segment / AUTO (Criteria A + B) ────────────── def step2(self, df): self.L("Step 2 — LOB eligibility …") b = len(df) lob = fc(df.columns, "lob"); sub = fc(df.columns, "sub_lob") seg = fc(df.columns, "segment"); ivt = fc(df.columns, "inv_type") if not lob: self.L(" WARNING: LOB column not found — skipping", "warning") return df ELIG = {"CBD", "NEW ENGINE", "REBUILD", "SOLUTIONS", "PARTS"} df = df[N(df[lob]).isin(ELIG)].copy() self.L(f" After LOB filter: {len(df):,}") if sub: pts = N(df[lob]) == "PARTS" sok = N(df[sub]).isin({"CORE", "FILTERS", "RECON", ""}) df = df[~(pts & ~sok)].copy() if seg: df = df[N(df[seg]) != "EDO"].copy() if seg and ivt: auto = N(df[seg]) == "AUTO" css = N(df[ivt]).str.contains("CSS CORE COMP", regex=False, na=False) df = df[~(auto & ~css)].copy() self.L(f" Removed {b - len(df):,}, remain {len(df):,}", "success") return df # ── Step 3: DDS-Acoustics + Battery (Criteria C + D) ───────────────────── def step3(self, df): self.L("Step 3 — DDS-Acoustics & Battery …") b = len(df) bc = fc(df.columns, "batch_src"); ic = fc(df.columns, "item") if bc: df = df[~N(df[bc]).str.contains("DDS", regex=False, na=False)].copy() if ic: df = df[~(N(df[ic]) == "AX00000000")].copy() self.L(f" Removed {b - len(df):,}", "success") return df # ── Step 4: Credit Memo-Misc (Criteria E) ──────────────────────────────── def step4(self, df): self.L("Step 4 — Credit Memo-Misc …") b = len(df) bc = fc(df.columns, "batch_src"); lc = fc(df.columns, "line_item") if bc and lc: cm = (N(df[bc]).str.contains("CREDIT MEMO", regex=False, na=False) & N(df[bc]).str.contains("MISC", regex=False, na=False)) ln = N(df[lc]); ex = pd.Series(False, index=df.index) for kw in ["EARLY PAYMENT DISCOUNT", "INCENTIVE TO ENGINEER"]: ex = ex | (cm & ln.str.contains(kw, regex=False, na=False)) df = df[~ex].copy() self.L(f" Removed {b - len(df):,}", "success") return df # ── Step 5: Exclusion list (49 criteria) ───────────────────────────────── def step5(self, df): self.L("Step 5 — Exclusion list …") b = len(df) ivt = fc(df.columns, "inv_type") lc = fc(df.columns, "line_item") ic = fc(df.columns, "item") ex = self.excl.copy() at = next((c for c in ex.columns if any(k in c.lower() for k in ["apply", "column", "filter"])), ex.columns[-1]) atn = N(ex[at]) # ── Step 7: Finalise eligible records ──────────────────────────────────── # After all exclusions, remaining rows ARE eligible. # Only remove rows explicitly flagged NO — never require YES. def step7(self, df): self.L("Step 7 — Eligible records …") vd = fc(df.columns, "consider_vd") if vd: vn = N(df[vd]) NO_SET = {"NO", "N", "NOT ELIGIBLE", "INELIGIBLE", "0"} ex_no = vn.isin(NO_SET) if 0 < ex_no.sum() < len(df): df = df[~ex_no].copy() self.L(f" Explicit NO removed: {ex_no.sum():,}") elif ex_no.sum() == len(df): self.L(" WARNING: All rows flagged NO — skipping filter", "warning") rv = fc(df.columns, "revenue") self.stat["elig"] = len(df) self.stat["excl"] = self.stat.get("total", 0) - len(df) self.stat["sales_cr"] = to_num_safe(df[rv]).sum() / 1e7 if rv else 0.0 self.L(f" Eligible: {len(df):,} | ₹{self.stat['sales_cr']:,.2f} Cr", "success") return df #
