import subprocess import threading import webbrowser import time import sys import os def get_base_path(): """Get the base path — works both in .exe and normal Python.""" if getattr(sys, 'frozen', False): return sys._MEIPASS return os.path.dirname(os.path.abspath(__file__)) def open_browser(): """Wait for Streamlit to start, then open browser.""" time.sleep(4) webbrowser.open("http://localhost:8501") def main(): base = get_base_path() app_path = os.path.join(base, "app.py") # Start browser in background thread t = threading.Thread(target=open_browser, daemon=True) t.start() # Start Streamlit subprocess.run([ sys.executable, "-m", "streamlit", "run", app_path, "--server.port", "8501", "--server.headless", "true", "--browser.gatherUsageStats", "false", "--server.fileWatcherType", "none", ]) if __name__ == "__main__": main()
