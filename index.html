# processor.py | VLDS Processing Pipeline — Forest Dealerships # ───────────────────────────────────────────────────────────────────────────── # Pipeline: # 1 Load VLDS Q4 main file # 2 Filter by 4S Dealer List (customer numbers) # 3 LOB eligibility — only PARTS gets VLDS; others → "No-Other LOB" in Remark # 4 Apply exclusion list — each entry marks matching rows in Remark column # 5 DF&RECD exclusion — concat key match → "No-DF/RECD" in Remark # 6 Eligible = rows with BLANK Remark (>300 Cr) # 7 Quarterly Pivot — Customer × Fiscal Month × Revenue Amount # 8 Final Summary — merge Q4-24 from prev year file, calculate formulas # 9 Excel output — 3 sheets: Pivot, Final Summary, Eligible Data # ───────────────────────────────────────────────────────────────────────────── import pandas as pd import numpy as np from io import BytesIO from datetime import datetime import traceback import re import xlsxwriter from utils import (read_file, find_col, fcol, N, to_num, safe_num, quarter_from_filename, year_from_filename) class VLDSProcessor: def __init__(self, excl_df: pd.DataFrame, slab_df: pd.DataFrame): self.excl = excl_df self.slab = slab_df self.log = [] self.stats = {} def L(self, msg, lv="info"): self.log.append({"lv": lv, "msg": msg}) # ─── Step 1: Load VLDS main file ───────────────────────────────────────── def step1_load(self, vlds_f): self.L("Step 1 ▸ Loading VLDS Q4 2025 main file …") df = read_file(vlds_f, sheet_hints=["vlds","4s","main","data","q4"]) if df is None or df.empty: raise ValueError("VLDS Q4 2025 file is empty or could not be read.") self.stats["total"] = len(df) self.L(f" {len(df):,} rows | {len(df.columns)} columns") self.L(" All columns: " + " | ".join(df.columns.tolist())) # Log key column detections checks = { "LOB": ["lob","line of business"], "Remark": ["remark","remarks","consider for vd"], "Batch Source":["batch source"], "Invoice Type":["invoice type","order header subtype code"], "Line Item": ["line item","inventory item description","line item description"], "Item Number": ["item number","item"], "Revenue Amt": ["revenue amount"], "Invoice No": ["invoice number","invoice no"], "Cust Number": ["customer account number","customer number","dealer number"], "Cust Name": ["customer account name","customer name"], "Fiscal Month":["reporting to fiscal month","fiscal month","cmi month"], "Zone": ["zone"], "BD": ["bd"], } for label, hints in checks.items(): c = fcol(df.columns, *hints) if c: self.L(f" ✔ {label:15s} → '{c}'", "success") else: self.L(f" ✗ {label:15s} — NOT FOUND (check column names)", "warning") return df # ─── Step 2: Filter by 4S Dealer List ──────────────────────────────────── # Keep only rows where Customer Account Number is in the dealer list def step2_dealer_filter(self, df, dealer_f): self.L("Step 2 ▸ 4S Dealer List filter …") if dealer_f is None: self.L(" No dealer list uploaded — skipping", "warning") return df dl = read_file(dealer_f) if dl is None or dl.empty: self.L(" Dealer list empty — skipping", "warning") return df # Find customer number column in dealer file dn_col = (fcol(dl.columns, "customer number","dealer number", "customer account number","cust no","account number") or dl.columns[0]) dealer_nos = set(N(dl[dn_col]).tolist()) - {"", "NAN"} self.L(f" {len(dealer_nos)} dealer customer numbers loaded") # Find matching column in VLDS file vn_col = fcol(df.columns, "customer account number","customer number", "dealer number","cust no","account number") if not vn_col: self.L(" Customer Number column not found in VLDS — skipping filter", "warning") return df before = len(df) mask = N(df[vn_col]).isin(dealer_nos) df = df[mask].copy() self.L(f" After dealer filter: {before:,} → {len(df):,} rows " f"(₹{to_num(df[fcol(df.columns,'revenue amount','revenue_amount') or df.columns[0]]).sum()/1e7:,.1f} Cr)" if fcol(df.columns,"revenue amount","revenue_amount") else f" After dealer filter: {before:,} → {len(df):,} rows", "success") return df # ─── Step 3: LOB eligibility ────────────────────────────────────────────── # PARTS (Highway) → eligible for VLDS # CBD / New Engine / Rebuild / Solutions → "No-Other LOB" in Remark def step3_lob(self, df): self.L("Step 3 ▸ LOB eligibility filter …") lob_col = fcol(df.columns, "lob","line of business") rem_col = fcol(df.columns, "remark","remarks","consider for vd", "consider for vlds","eligible sales/ not eligible") if not lob_col: self.L(" LOB column not found — skipping", "warning") return df df = df.copy() # Ensure Remark column exists if not rem_col: df["Remark"] = "" rem_col = "Remark" lob_n = N(df[lob_col]) # LOBs that get VLDS: PARTS only (highway = parts line) ELIGIBLE_LOB = {"PARTS"} # LOBs explicitly listed but NOT eligible (get "No-Other LOB") NAMED_NON_ELIGIBLE = {"CBD", "NEW ENGINE", "REBUILD", "SOLUTIONS", "HIGHWAY"} already_flagged = N(df[rem_col]) != "" # Flag non-eligible named LOBs flag_named = lob_n.isin(NAMED_NON_ELIGIBLE) & ~already_flagged df.loc[flag_named, rem_col] = "No-Other LOB" # Flag everything else that is NOT PARTS and NOT already flagged flag_rest = (~lob_n.isin(ELIGIBLE_LOB | NAMED_NON_ELIGIBLE)) & ~already_flagged & ~flag_named df.loc[flag_rest, rem_col] = "No-Other LOB" parts_ct = (lob_n == "PARTS").sum() flagged = (flag_named | flag_rest).sum() self.L(f" PARTS rows: {parts_ct:,} | Non-eligible flagged: {flagged:,}", "success") return df # ─── Step 4: Apply exclusion list ───────────────────────────────────────── # Each exclusion list row: # Particulars = the value to search for # Column = which column to search in (Invoice Type / Line Item / Batch Source / etc.) # Item No = if not NA, also check Item Number column # Matching rows get "No-" written into the Remark column. def step4_exclusions(self, df): self.L("Step 4 ▸ Applying exclusion list …") rem_col = fcol(df.columns, "remark","remarks","consider for vd") or "Remark" inv_col = fcol(df.columns, "invoice type","invoicetype", "order header subtype code","transaction type") line_col = fcol(df.columns, "line item","line item description", "inventory item description","item description") item_col = fcol(df.columns, "item number","item_number","item") batch_col= fcol(df.columns, "batch source","batchsource") seg_col = fcol(df.columns, "segment") df = df.copy() if rem_col not in df.columns: df[rem_col] = "" total_flagged = 0 for _, row in self.excl.iterrows(): part = str(row.get("Particulars","")).strip() item_n = str(row.get("Item No","NA")).strip().upper() col_hint= str(row.get("Column","Line Item")).strip().upper() if not part: continue part_upper = part.upper() # Map "Column" field to actual DataFrame column if "INVOICE" in col_hint or "TRANSACTION" in col_hint: target = inv_col elif "BATCH" in col_hint: target = batch_col elif "SEGMENT" in col_hint: target = seg_col else: target = line_col # default: line item description already = N(df[rem_col]) != "" mask = pd.Series(False, index=df.index) # Search in target column if target and target in df.columns: mask = mask | N(df[target]).str.contains(part_upper, regex=False, na=False) # Cross-check item number if specified if item_n not in ("NA", "N/A", "") and item_col and item_col in df.columns: mask = mask | (N(df[item_col]) == item_n) new_flags = mask & ~already if new_flags.sum() > 0: df.loc[new_flags, rem_col] = f"No-{part}" total_flagged += new_flags.sum() self.L(f" Exclusion list flagged {total_flagged:,} rows across {len(self.excl)} criteria", "success") return df # ─── Step 5: DF&RECD exclusion ─────────────────────────────────────────── # Build concat key from DF&RECD file: InvoiceNo + ItemNo + RevenueAmount # Match against BD column in VLDS file → "No-DF/RECD" in Remark def step5_dfrecd(self, df, dfrecd_f): self.L("Step 5 ▸ DF&RECD exclusion …") rem_col = fcol(df.columns, "remark","remarks","consider for vd") or "Remark" bd_col = fcol(df.columns, "bd") inv_col = fcol(df.columns, "invoice number","invoice no","invoice_number") item_col = fcol(df.columns, "item number","item_number","item") rev_col = fcol(df.columns, "revenue amount","revenue_amount") df = df.copy() if rem_col not in df.columns: df[rem_col] = "" quarter = "Q4" recd_keys: set = set() if dfrecd_f is not None: try: quarter = quarter_from_filename(dfrecd_f.name) yr = year_from_filename(dfrecd_f.name) self.L(f" DF&RECD file: quarter={quarter}, year={yr}") self.stats["quarter"] = quarter self.stats["year"] = yr dr = read_file(dfrecd_f) if dr is not None and not dr.empty: self.L(f" DF&RECD: {len(dr):,} rows loaded") d_inv = fcol(dr.columns, "invoice number","invoice no") or dr.columns[0] d_item = fcol(dr.columns, "item number","item") d_rev = fcol(dr.columns, "revenue amount","revenue_amount") # Build concat key: Invoice No + Item No + Revenue Amount inv_s = dr[d_inv].fillna("").astype(str).str.strip() item_s = dr[d_item].fillna("").astype(str).str.strip() if d_item else "" rev_s = dr[d_rev].fillna("").astype(str).str.strip() if d_rev else "" recd_keys = set((inv_s + item_s + rev_s).tolist()) recd_keys.discard("") self.L(f" {len(recd_keys):,} DF&RECD concat keys built") except Exception as e: self.L(f" WARNING: DF&RECD file error: {e}", "warning") already = N(df[rem_col]) != "" # Method A: match via BD column (pre-computed concatenation in VLDS) matched_bd = 0 if bd_col and bd_col in df.columns and recd_keys: bd_vals = df[bd_col].fillna("").astype(str).str.strip() bd_mask = bd_vals.isin(recd_keys) & ~already df.loc[bd_mask, rem_col] = "No-DF/RECD" matched_bd = bd_mask.sum() self.L(f" BD column matches: {matched_bd:,}") already = N(df[rem_col]) != "" # Method B: build concat key from VLDS columns and match matched_key = 0 if recd_keys and inv_col and rev_col: inv_s = df[inv_col].fillna("").astype(str).str.strip() item_s = df[item_col].fillna("").astype(str).str.strip() if item_col else "" rev_s = df[rev_col].fillna("").astype(str).str.strip() vlds_key = inv_s + item_s + rev_s key_mask = vlds_key.isin(recd_keys) & ~already df.loc[key_mask, rem_col] = "No-DF/RECD" matched_key = key_mask.sum() self.L(f" Concat key matches: {matched_key:,}") self.L(f" DF&RECD total flagged: {matched_bd + matched_key:,}", "success") self.stats["quarter"] = self.stats.get("quarter", quarter) return df # ─── Step 6: Identify eligible records ─────────────────────────────────── # Eligible = rows with BLANK Remark (no exclusion was applied) def step6_eligible(self, df): self.L("Step 6 ▸ Identifying eligible records (blank Remark) …") rem_col = fcol(df.columns, "remark","remarks","consider for vd") if rem_col: rm_norm = N(df[rem_col]) eligible = df[rm_norm == ""].copy() else: # No remark column found — treat all remaining as eligible eligible = df.copy() self.L(" Remark column not found — all rows treated as eligible", "warning") rev_col = fcol(eligible.columns, "revenue amount","revenue_amount") sales_cr = to_num(eligible[rev_col]).sum() / 1e7 if rev_col else 0.0 self.stats["eligible"] = len(eligible) self.stats["excluded"] = self.stats.get("total", 0) - len(eligible) self.stats["sales_cr"] = sales_cr self.L(f" Eligible: {len(eligible):,} rows | ₹{sales_cr:,.2f} Cr", "success") return eligible # ─── Step 7: Quarterly Pivot ────────────────────────────────────────────── # Rows: Customer Account Name + Customer Account Number (+Zone if present) # Cols: Fiscal Month (e.g. 2025/10, 2025/11, 2025/12) # Values: Sum of Revenue Amount # Extra: Grand Total column = row-wise sum def step7_pivot(self, eligible): self.L("Step 7 ▸ Building quarterly pivot …") cn_col = fcol(eligible.columns, "customer account name","customer name") no_col = fcol(eligible.columns, "customer account number","customer number", "dealer number") rev_col = fcol(eligible.columns, "revenue amount","revenue_amount") mo_col = fcol(eligible.columns, "reporting to fiscal month", "reporting fiscal month name","fiscal month","cmi month","month") zn_col = fcol(eligible.columns, "zone","region") if not rev_col: self.L(" Revenue Amount column not found — cannot build pivot", "warning") return pd.DataFrame({"Error": ["Revenue Amount column not found"]}) if not cn_col and not no_col: self.L(" Customer Name/Number column not found — cannot build pivot", "warning") return pd.DataFrame({"Error": ["Customer column not found"]}) elig = eligible.copy() elig[rev_col] = to_num(elig[rev_col]) # strip commas, convert to float # Pivot index: always include name + number; add zone if present idx = list(dict.fromkeys([c for c in [cn_col, no_col, zn_col] if c])) if mo_col: months = sorted(elig[mo_col].dropna().astype(str).str.strip().unique()) self.L(f" Fiscal months found: {months}") try: elig["_mo"] = elig[mo_col].fillna("Unknown").astype(str).str.strip() pv = elig.pivot_table(values=rev_col, index=idx, columns="_mo", aggfunc="sum", fill_value=0).reset_index() pv.columns.name = None elig.drop(columns=["_mo"], errors="ignore", inplace=True) mc = [c for c in pv.columns if c not in idx] for c in mc: pv[c] = pd.to_numeric(pv[c], errors="coerce").fillna(0.0) pv["Grand Total"] = pv[mc].sum(axis=1) self.L(f" Pivot: {len(pv)} customers × {len(mc)} months", "success") return pv except Exception as e: self.L(f" Pivot error ({e}) — totals fallback", "warning") # Fallback: totals only grp = elig.groupby(idx)[rev_col].sum().reset_index() grp.rename(columns={rev_col: "Current Quarter Sales"}, inplace=True) self.L(f" Customer totals (no month column): {len(grp)} rows", "success") return grp # ─── Step 8: Final Summary ──────────────────────────────────────────────── # Merge Q4-24 data from previous year file using user-selected columns. # Formulas: # Increase = Q4-25 Sales - Q4-24 Sales # Growth % = Increase / Q4-24 Sales × 100 # Discount % = from slab table based on Growth % # Discount Value= Q4-25 Sales × Discount % / 100 def step8_summary(self, pivot, prev_f, prev_join_col=None, prev_amt_col=None, slab_type="quarterly"): self.L("Step 8 ▸ Building Final Summary …") quarter = self.stats.get("quarter", "Q4") year = self.stats.get("year", str(datetime.now().year)) yr2 = year[-2:] # e.g. "25" curr_lbl= f"{quarter}-{yr2}" # e.g. "Q4-25" prev_lbl= f"{quarter}-{str(int(yr2)-1).zfill(2)}" # e.g. "Q4-24" # Identify ID columns in pivot cn = next((c for c in pivot.columns if any(k in c.lower() for k in ["account name","customer name","customername"])), None) co = next((c for c in pivot.columns if any(k in c.lower() for k in ["account number","customer number","customer_number", "cust no","dealer number"])), None) zn = next((c for c in pivot.columns if c.lower() in ["zone","region","territory"]), None) curr_col = ("Grand Total" if "Grand Total" in pivot.columns else "Current Quarter Sales" if "Current Quarter Sales" in pivot.columns else pivot.columns[-1]) sm = pivot.copy() sm[curr_col] = pd.to_numeric( sm[curr_col].astype(str).str.replace(",","",regex=False), errors="coerce").fillna(0.0) sm[f"Final Amount {curr_lbl}"] = sm[curr_col] # Zone if zn and zn != "Zone": sm["Zone"] = sm[zn].fillna("").astype(str) elif "Zone" not in sm.columns: sm["Zone"] = "" sm[prev_lbl] = 0.0 # will be populated from prev year file # ── Merge previous year data ────────────────────────────── if prev_f: try: pf = read_file(prev_f) if pf is None or pf.empty: self.L(f" WARNING: Previous year file is empty", "warning") else: self.L(" Prev year file columns: " + " | ".join(pf.columns.tolist())) self.L(f" Prev year rows: {len(pf)}") # Show first row sample if len(pf) > 0: s = {c: str(pf.iloc[0][c])[:20] for c in pf.columns[:6]} self.L(f" First row: {s}") # ── Determine join column ───────────────────── if prev_join_col and prev_join_col in pf.columns: jc = prev_join_col self.L(f" Using USER-SELECTED join col: '{jc}'") else: jc = (fcol(pf.columns,"customer account number","customer number", "dealer number","cust no") or fcol(pf.columns,"customer account name","customer name") or pf.columns[0]) self.L(f" Auto-detected join col: '{jc}'") # ── Determine amount column ─────────────────── if prev_amt_col and prev_amt_col in pf.columns: pa = prev_amt_col self.L(f" Using USER-SELECTED amount col: '{pa}'") else: pa = None col_lc = {c.lower().strip(): c for c in pf.columns} for nm in [prev_lbl.lower(), "q4-24","q4 24","q4_24", "final amount q4-24","q4-2024", "previous quarter","prev quarter"]: if nm in col_lc: pa = col_lc[nm]; break if not pa: for c in pf.columns: if "q4" in c.lower() or "quarter" in c.lower(): nc = pd.to_numeric( pf[c].astype(str).str.replace(",","",regex=False), errors="coerce").notna().sum() if nc > len(pf)*0.3: pa=c; break if not pa: # Highest numeric density after col 3 best_c, best_s = None, 0 for c in pf.columns[3:]: sc = pd.to_numeric( pf[c].astype(str).str.replace(",","",regex=False), errors="coerce").notna().mean() if sc > best_s: best_s, best_c = sc, c if best_c and best_s > 0.3: pa = best_c self.L(f" Auto-detected amount col: '{pa}'") if pa: # Log sample values for debugging jc_samp = pf[jc].dropna().astype(str).str.strip().unique()[:5].tolist() pa_samp = pf[pa].dropna().unique()[:4].tolist() self.L(f" Join col '{jc}' sample: {jc_samp}") self.L(f" Amount col '{pa}' sample: {pa_samp}") # Build lookup: sum amount per join-key pf_w = pf.copy() pf_w["_amt"] = pf_w[pa].apply( lambda x: safe_num(str(x).replace(",",""))) pg = pf_w.groupby(jc)["_amt"].sum().reset_index() pg.columns = ["_jk", prev_lbl] # Zone from prev file pz = fcol(pf.columns,"zone","region") if pz: zg = pf_w[[jc, pz]].drop_duplicates(subset=[jc]) zg.columns = ["_jk","_zp"] pg = pg.merge(zg, on="_jk", how="left") extra = ["_zp"] if "_zp" in pg.columns else [] # ── Merge: try each identifier column ──── merged = False for pcol in [co, cn]: if not pcol or pcol not in sm.columns: continue # Normalise: strip + upper smk = sm[pcol].fillna("").astype(str).str.strip().str.upper() pgk = pg["_jk"].fillna("").astype(str).str.strip().str.upper() sm_n = sm.copy(); sm_n["_nk"] = smk pg_n = pg.copy(); pg_n["_nk"] = pgk t = sm_n.merge(pg_n[["_nk",prev_lbl]+extra], on="_nk", how="left", suffixes=("","_r")) matched = int(t[prev_lbl].notna().sum()) self.L(f" Exact merge on '{pcol}': {matched}/{len(sm)} matched") # Numeric-only fallback (strip non-digits) if matched == 0: sm_n2 = sm.copy() pg_n2 = pg.copy() sm_n2["_nk"] = smk.apply( lambda x: "".join(c for c in x if c.isdigit())) pg_n2["_nk"] = pgk.apply( lambda x: "".join(c for c in x if c.isdigit())) t2 = sm_n2.merge(pg_n2[["_nk",prev_lbl]+extra], on="_nk", how="left", suffixes=("","_r")) m2 = int(t2[prev_lbl].notna().sum()) if m2 > matched: self.L(f" Numeric fallback on '{pcol}': {m2} matched") t, matched = t2, m2 # Show unmatched samples if matched < len(sm) and "_nk" in t.columns: un = t.loc[t[prev_lbl].isna(),"_nk"].unique()[:4].tolist() self.L(f" Unmatched pivot keys (sample): {un}") if matched > 0: sm = t.copy() sm[prev_lbl] = sm[prev_lbl].apply( lambda x: safe_num(str(x)) if pd.notna(x) else 0.0) if "_zp" in sm.columns: blank = sm["Zone"].fillna("").str.strip() == "" sm.loc[blank,"Zone"] = sm.loc[blank,"_zp"].fillna("") sm.drop(columns=["_zp"], inplace=True) drop = [c for c in ["_nk","_jk",f"{prev_lbl}_r"] if c in sm.columns] sm.drop(columns=drop, inplace=True) self.L(f" ✔ {prev_lbl} merged on '{pcol}': " f"{matched} rows populated", "success") merged = True break if not merged: self.L( f" ✘ {prev_lbl} merge failed — 0 rows matched.\n" f" ACTION: Use the column mapping dropdowns above the Run\n" f" button to manually select the Customer Identifier and\n" f" {prev_lbl} Sales Amount columns from the prev year file.", "warning") else: self.L(f" {prev_lbl} amount column not found in previous year file. " "Select it manually using the dropdown.", "error") except Exception as e: self.L(f" ERROR in prev year merge: {e}", "error") self.L(" " + traceback.format_exc(), "error") # ── Calculations ────────────────────────────────────────── payout_col = ("Quarterly Payout(%)" if "quarterly" in slab_type.lower() else "Annual Payout(%)") q425 = sm[f"Final Amount {curr_lbl}"].apply(safe_num) q424 = sm[prev_lbl].apply(lambda x: safe_num(str(x)) if pd.notna(x) else 0.0) sm[f"Final Amount {curr_lbl}"] = q425 sm[f"Final Amount {prev_lbl}"] = q424 sm["Increase"] = q425 - q424 prev_safe = q424.replace(0, np.nan) sm["Growth %"] = ((sm["Increase"] / prev_safe * 100) .replace([np.inf,-np.inf],0).fillna(0)) sm["Discount %"] = sm["Growth %"].apply( lambda g: self._slab_lookup(g, payout_col)) sm["Discount Value"] = (q425 * sm["Discount %"] / 100).fillna(0) # Sanitise for c in sm.select_dtypes(include=[np.number]).columns: sm[c] = sm[c].replace([np.inf,-np.inf],0).fillna(0) # ── Final column order ──────────────────────────────────── if zn and zn in sm.columns and zn != "Zone": sm.rename(columns={zn:"Zone"}, inplace=True) id_cols = [c for c in [cn,"Zone",co] if c and c in sm.columns] calc_cols = [f"Final Amount {curr_lbl}", prev_lbl, f"Final Amount {prev_lbl}", "Increase","Growth %","Discount %","Discount Value"] out_cols = id_cols + [c for c in calc_cols if c in sm.columns] sm = sm[out_cols].copy() # Clean header names rn = {} if cn and cn in sm.columns: rn[cn] = "Customer Name" if co and co in sm.columns: rn[co] = "Customer Number" sm.rename(columns=rn, inplace=True) self.L(f" Final Summary: {len(sm)} rows, cols: {list(sm.columns)}", "success") return sm, curr_lbl def _slab_lookup(self, growth_pct: float, payout_col: str) -> float: g = safe_num(growth_pct) for _, r in self.slab.iterrows(): try: lo = float(r.get("Growth Min(%)", 0) or 0) hi = float(r.get("Growth Max(%)", 9999) or 9999) if lo <= g < hi: return float(r.get(payout_col, r.get("Quarterly Payout(%)", 0)) or 0) except Exception: continue return 0.0 #
