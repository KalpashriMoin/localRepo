def step4_exclusions(self, df): self.L("Step 4 ▸ Applying exclusion list …") rem_col = fcol(df.columns, "remark","remarks","consider for vd") or "Remark" inv_col = fcol(df.columns, "invoice type","invoicetype", "order header subtype code","transaction type") line_col = fcol(df.columns, "line item","line item description", "inventory item description","item description") item_col = fcol(df.columns, "item number","item_number","item no","item", "inventory item number") batch_col = fcol(df.columns, "batch source","batchsource") seg_col = fcol(df.columns, "segment") df = df.copy() if rem_col not in df.columns: df[rem_col] = "" total_f = 0 for _, row in self.excl.iterrows(): part = str(row.get("Particulars","")).strip() item_n = str(row.get("Item No","NA")).strip().upper() col_hint = str(row.get("Column","Line Item")).strip().upper() if not part: continue pu = part.upper() # Determine which VLDS column to search the Particulars value in # based on "Column where filter applied" tgt = None if "INVOICE" in col_hint or "TRANSACTION" in col_hint: tgt = inv_col elif "BATCH" in col_hint: tgt = batch_col elif "SEGMENT" in col_hint: tgt = seg_col elif "LINE ITEM" in col_hint or "LINE" in col_hint: tgt = line_col elif "ITEM" in col_hint: tgt = item_col else: tgt = line_col # default fallback already = N(df[rem_col]) != "" mask = pd.Series(False, index=df.index) if item_n in ("NA", "N/A", ""): # Item Name is NA → search Particulars value in the column # specified by "Column where filter applied" if tgt and tgt in df.columns: mask |= N(df[tgt]).str.contains(pu, regex=False, na=False) else: # Item Name has a value → search Item Name in Item column # AND search Particulars in Line Item column if item_col and item_col in df.columns: mask |= (N(df[item_col]).str.contains(item_n, regex=False, na=False)) if line_col and line_col in df.columns: mask |= N(df[line_col]).str.contains(pu, regex=False, na=False) nf = mask & ~already if nf.sum() > 0: df.loc[nf, rem_col] = f"Exclusion List - {part}" total_f += nf.sum() self.L(f" Excluded {nf.sum():,} rows for: {part}") self.L(f" Total exclusion flagged: {total_f:,} rows", "success") return df
