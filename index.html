""" import io import hashlib import pandas as pd import streamlit as st # ── Sheet names ─────────────────────────────────────────────────────────────── def get_sheet_names(raw: bytes, filename: str) -> list: try: nm = filename.lower() if nm.endswith(".xlsb"): import pyxlsb with pyxlsb.open_workbook(io.BytesIO(raw)) as wb: return list(wb.sheets) if nm.endswith(".csv"): return [] return pd.ExcelFile(io.BytesIO(raw), engine="openpyxl").sheet_names except Exception: return [] #
