# utils.py | VLDS Pro v2 — Forest Dealerships import re import pandas as pd import numpy as np # ── Key normaliser ──────────────────────────────────────────────────────────── # Excel reads pure integers as floats → str gives "12345.0". # _norm_key strips ".0" so cross-file joins always work. def _norm_key(v): s = str(v).strip() if re.match(r"^\d+\.0+$", s): s = s.split(".")[0] return s # ── Column finder ───────────────────────────────────────────────────────────── def fcol(columns, *hints): """Return the first column name whose lowercase contains any hint.""" for h in hints: for c in columns: if h.lower() in str(c).lower(): return c return None # ── Normalised string series ────────────────────────────────────────────────── def N(s): """Fillna + astype str + strip + upper — for consistent matching.""" return s.fillna("").astype(str).str.strip().str.upper() # ── Numeric conversion ──────────────────────────────────────────────────────── def to_num(s): """Series → float64: strip commas, coerce, NaN→0.""" return pd.to_numeric( s.astype(str).str.replace(",", "", regex=False).str.strip(), errors="coerce").fillna(0.0) def safe_num(v): """Single value → float, 0.0 on failure.""" try: f = float(str(v).replace(",", "").strip()) return 0.0 if (pd.isna(f) or np.isinf(f)) else f except Exception: return 0.0 # ── Quarter / year from filename ────────────────────────────────────────────── def quarter_from_filename(name): m = re.search(r"Q([1-4])", str(name), re.IGNORECASE) return f"Q{m.group(1).upper()}" if m else "Q4" def year_from_filename(name): m = re.search(r"(20\d{2})", str(name)) return m.group(1) if m else str(pd.Timestamp.now().year) # ── Smart file reader ───────────────────────────────────────────────────────── def read_file(f, sheet_hints=None): """ Read an uploaded Excel or CSV file into a clean DataFrame. their updated file. Column matching is by keyword, not exact name. def parse_excl(f) -> pd.DataFrame: """ Read the Exclusion List file dynamically. Handles your actual slab file column names: 'Growth over 2024' → growth range (min/max) 'Incentive Payout' → quarterly/annual payout rates Plus all standard naming variants. """ EMPTY = pd.DataFrame(columns=["Growth Min(%)", "Growth Max(%)", "Quarterly Payout(%)", "Annual Payout(%)"]) if df is None or df.empty: return EMPTY cols = list(df.columns) cl = {str(c).lower().strip(): c for c in cols} def _find(*hints): """Find column by exact then substring match.""" for h in hints: h = h.lower().strip() if h in cl: return cl[h] for h in hints: h = h.lower().strip() for k, v in cl.items(): if h and k and (h in k or k in h): return v return None # ── Growth range columns ────────────────────────────────────────────────── # Your file: "Growth over 2024" — first occurrence is min, second is max # Also handle: "Growth Min(%)", "Min", "From", "Slab From" etc. growth_cols = [c for c in cols if any(k in str(c).lower() for k in ["growth over","growth min","growth max","growth %", "growth(%","growth ("])] min_c = _find("growth min(%)","growth min","growth_min", "growth over 2024","growth over","min(%)","min %","min","from","lower","start") max_c = _find("growth max(%)","growth max","growth_max","max(%)","max %","max","to","upper","end") # Two "growth" cols → first=min, second=max min_c = growth_cols[0] max_c = growth_cols[1] # ── Payout columns ──────────────────────────────────────────────────────── # Your file: "Incentive Payout" with sub-cols for quarterly/annual # Look for: "quarterly", "q pay", "incentive payout", "payout" etc. q_c = _find("quarterly payout(%)","quarterly payout","quarterly %", "q payout","quarterly","qtr","q pay", "incentive payout quarterly","incentive quarterly") a_c = _find("annual payout(%)","annual payout","annual %", "a payout","annual","yearly","a pay", "incentive payout annual","incentive annual") # If payout cols not found by name, look for numeric columns after the growth cols # and use positional: 3rd numeric col = quarterly, 4th = annual if not q_c or not a_c: # Find all columns that are primarily numeric (not the growth/min/max ones) used = {min_c, max_c} numeric_leftover = [] for c in cols: if c in used: continue vals = pd.to_numeric(df[c].astype(str).str.replace(",","",regex=False), errors="coerce") if vals.notna().mean() >= 0.5: #
