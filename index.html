═════════════════════════════════════════════════════════════════════════════ # COLUMN FINDER — 3-pass: exact, contains, partial # ═════════════════════════════════════════════════════════════════════════════ def fc(df, *candidates): """Find column — exact match first, then contains, then partial.""" if df is None or df.empty: return None lk = {c.lower().strip(): c for c in df.columns} # Pass 3 partial reverse for cand in candidates: k = cand.lower().strip() for col_l, col_a in lk.items(): if col_l in k: return col_a return None def n(s): return pd.to_numeric(s, errors="coerce").fillna(0) def inr(v): try: f = float(v) return "₹0.00" if (np.isnan(f) or np.isinf(f)) else f"₹{f:,.2f}" except Exception: return "₹0.00" def sv(v): if v is None: return "" if isinstance(v, float) and (np.isnan(v) or np.isinf(v)): return "" return v #
